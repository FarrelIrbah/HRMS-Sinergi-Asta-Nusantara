```typescript
// ─── Seed: Employee Salaries ─────────────────────────────────────────
console.log("Seeding employee salaries...");
// Update all employees with a default baseSalary if they have none set
await prisma.employee.updateMany({
  where: { baseSalary: 0 },
  data: { baseSalary: 5000000 },
});
// Give a few employees distinct salaries for meaningful payroll test data
const salaryOverrides = [
  { nik: "EMP-2024-0001", baseSalary: 8000000 },
  { nik: "EMP-2024-0002", baseSalary: 6500000 },
  { nik: "EMP-2024-0003", baseSalary: 7200000 },
];
for (const override of salaryOverrides) {
  const emp = await prisma.employee.findFirst({ where: { nik: override.nik } });
  if (emp) {
    await prisma.employee.update({
      where: { id: emp.id },
      data: { baseSalary: override.baseSalary },
    });
    // Add a transport allowance if none exists
    const existing = await prisma.employeeAllowance.findFirst({ where: { employeeId: emp.id } });
    if (!existing) {
      await prisma.employeeAllowance.create({
        data: {
          employeeId: emp.id,
          name: "Tunjangan Transport",
          amount: 500000,
          isFixed: true,
        },
      });
    }
  }
}
```
