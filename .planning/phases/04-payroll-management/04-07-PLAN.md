```typescript
import Decimal from "decimal.js";
import { differenceInMonths } from "date-fns";
import { Religion } from "@/types/enums";

export const RELIGION_HOLIDAY_MAP: Record<Religion, string> = {
  ISLAM: "Idul Fitri",
  KRISTEN: "Natal",
  KATOLIK: "Natal",
  HINDU: "Nyepi",
  BUDDHA: "Waisak",
  KONGHUCU: "Tahun Baru Imlek",
};

export interface THRInput {
  joinDate: Date;
  referenceDate: Date;  // The payroll run date (used for service length calculation)
  baseSalary: Decimal;
  fixedAllowancesTotal: Decimal;  // Only isFixed=true allowances per Permenaker 6/2016
  religion: Religion;
}

export interface THRResult {
  serviceMonths: number;
  isEligible: boolean;
  thrAmount: Decimal;
  holidayName: string;
  calculationNote: string;
}

/**
 * Calculate THR for a single employee per Permenaker No. 6/2016.
 *
 * Basis: gaji pokok + tunjangan tetap (isFixed=true allowances only).
 * Service months: differenceInMonths(referenceDate, joinDate) — floor of complete months.
 *
 * Rules:
 * - >= 12 months: 1 × (baseSalary + fixedAllowancesTotal)
 * - 1-11 months: (serviceMonths / 12) × 1 bulan gaji
 * - < 1 month: not eligible (amount = 0)
 */
export function calculateEmployeeTHR(input: THRInput): THRResult {
  const { joinDate, referenceDate, baseSalary, fixedAllowancesTotal, religion } = input;
  const monthlySalary = baseSalary.plus(fixedAllowancesTotal);
  const serviceMonths = differenceInMonths(referenceDate, joinDate);
  const holidayName = RELIGION_HOLIDAY_MAP[religion];

  if (serviceMonths < 1) {
    return {
      serviceMonths,
      isEligible: false,
      thrAmount: new Decimal(0),
      holidayName,
      calculationNote: "Masa kerja < 1 bulan — tidak berhak THR",
    };
  }

  if (serviceMonths >= 12) {
    return {
      serviceMonths,
      isEligible: true,
      thrAmount: monthlySalary.toDecimalPlaces(0),
      holidayName,
      calculationNote: `Masa kerja ${serviceMonths} bulan (≥ 12) — 1× gaji: ${monthlySalary.toFixed(0)}`,
    };
  }

  // Prorated: (serviceMonths / 12) × monthlySalary
  const thrAmount = monthlySalary
    .mul(serviceMonths)
    .dividedBy(12)
    .toDecimalPlaces(0);
  return {
    serviceMonths,
    isEligible: true,
    thrAmount,
    holidayName,
    calculationNote: `Masa kerja ${serviceMonths} bulan — ${serviceMonths}/12 × gaji: ${thrAmount.toFixed(0)}`,
  };
}
```

```typescript
export async function addTHRToPayrollAction(input: unknown) {
  await requireHRAdmin();
  const data = calculateTHRSchema.parse(input); // { month, year }

  // Find the DRAFT payroll run for this month/year
  const run = await prisma.payrollRun.findUnique({
    where: { month_year: { month: data.month, year: data.year } },
    include: { entries: { select: { id: true, employeeId: true } } }
  });

  if (!run) throw new Error("Jalankan penggajian untuk bulan ini terlebih dahulu");
  if (run.status === "FINALIZED") throw new Error("Payroll sudah difinalisasi");

  // Fetch all active employees with baseSalary, allowances, joinDate, religion
  const employees = await prisma.employee.findMany({
    where: { isActive: true },
    include: { allowances: { where: { isFixed: true } } },
  });

  // For each employee that has a payroll entry this month, update thrAmount
  const referenceDate = new Date(data.year, data.month - 1, 1);

  for (const emp of employees) {
    const entry = run.entries.find(e => e.employeeId === emp.id);
    if (!entry) continue;
    if (!emp.agama) continue;

    const fixedAllowancesTotal = emp.allowances.reduce(
      (sum, a) => sum.plus(new Decimal(a.amount.toString())),
      new Decimal(0)
    );
    const result = calculateEmployeeTHR({
      joinDate: emp.joinDate,
      referenceDate,
      baseSalary: new Decimal(emp.baseSalary.toString()),
      fixedAllowancesTotal,
      religion: emp.agama,
    });

    if (result.isEligible && result.thrAmount.gt(0)) {
      // Add THR to grossPay and recalculate netPay
      const currentEntry = await prisma.payrollEntry.findUnique({ where: { id: entry.id } });
      if (!currentEntry) continue;
      const newGross = new Decimal(currentEntry.grossPay.toString()).plus(result.thrAmount);
      const newNet = new Decimal(currentEntry.netPay.toString()).plus(result.thrAmount);
      // Note: PPh 21 is NOT recalculated here — THR tax effect handled in the next full payroll run
      await prisma.payrollEntry.update({
        where: { id: entry.id },
        data: {
          thrAmount: result.thrAmount.toNumber(),
          grossPay: newGross.toNumber(),
          netPay: newNet.toNumber(),
        },
      });
    }
  }

  revalidatePath("/payroll");
  revalidatePath("/payroll/thr");
  return { success: true };
}
```
