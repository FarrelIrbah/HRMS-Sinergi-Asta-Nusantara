async function canAccessEmployeeDocuments(session, employeeId): boolean {
  if (["HR_ADMIN", "SUPER_ADMIN"].includes(session.user.role)) return true;
  if (session.user.role === "MANAGER") {
    // Check if employee is in manager's department

  if (session.user.role === "EMPLOYEE") {
    const emp = await getEmployeeByUserId(session.user.id);

  return false;
}
