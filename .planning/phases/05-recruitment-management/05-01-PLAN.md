model Vacancy {
  id           String        @id @default(cuid())
  title        String
  departmentId String
  positionId   String?
  description  String?
  requirements String?
  openDate     DateTime      @db.Date
  status       VacancyStatus @default(BUKA)
  createdBy    String
  createdAt    DateTime      @default(now())
  updatedAt    DateTime      @updatedAt

  department Department  @relation(fields: [departmentId], references: [id])
  candidates Candidate[]

  @@map("vacancies")
}

model Candidate {
  id            String         @id @default(cuid())
  vacancyId     String
  namaLengkap   String
  email         String?
  nomorHp       String?
  stage         CandidateStage @default(MELAMAR)
  appliedAt     DateTime       @default(now())
  cvFileName    String?
  cvFilePath    String?
  cvMimeType    String?
  cvFileSize    Int?
  offeredSalary Decimal?       @db.Decimal(15, 2)
  startDate     DateTime?      @db.Date
  notes         String?
  isConverted   Boolean        @default(false)
  convertedAt   DateTime?
  convertedToId String?
  createdAt     DateTime       @default(now())
  updatedAt     DateTime       @updatedAt

  vacancy   Vacancy    @relation(fields: [vacancyId], references: [id], onDelete: Cascade)
  interview Interview?

  @@index([vacancyId, stage])
  @@map("candidates")
}

model Interview {
  id            String   @id @default(cuid())
  candidateId   String   @unique
  scheduledAt   DateTime
  notes         String?
  interviewerId String
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt

  candidate   Candidate @relation(fields: [candidateId], references: [id], onDelete: Cascade)
  interviewer User      @relation("InterviewerInterviews", fields: [interviewerId], references: [id])

  @@map("interviews")
}
```

Then run:

import { z } from "zod";

export const createVacancySchema = z.object({
  title: z.string().min(3, "Judul minimal 3 karakter").max(200),
  departmentId: z.string().min(1, "Departemen wajib dipilih"),
  positionId: z.string().optional(),
  description: z.string().optional(),
  requirements: z.string().optional(),
  openDate: z.string().min(1, "Tanggal buka wajib diisi"), // ISO date string from date input
});
export type CreateVacancyInput = z.infer<typeof createVacancySchema>;

export const updateVacancyStatusSchema = z.object({
  vacancyId: z.string().min(1),
  status: z.enum(["BUKA", "TUTUP"]),
});
export type UpdateVacancyStatusInput = z.infer<typeof updateVacancyStatusSchema>;

export const createCandidateSchema = z.object({
  vacancyId: z.string().min(1),
  namaLengkap: z.string().min(2, "Nama minimal 2 karakter").max(200),
  email: z.string().email("Email tidak valid").optional().or(z.literal("")),
  nomorHp: z.string().optional(),
});
export type CreateCandidateInput = z.infer<typeof createCandidateSchema>;

export const updateCandidateStageSchema = z.object({
  candidateId: z.string().min(1),
  stage: z.enum(["MELAMAR", "SELEKSI_BERKAS", "INTERVIEW", "PENAWARAN", "DITERIMA", "DITOLAK"]),
});
export type UpdateCandidateStageInput = z.infer<typeof updateCandidateStageSchema>;

export const scheduleInterviewSchema = z.object({
  candidateId: z.string().min(1),
  scheduledAt: z.string().min(1, "Tanggal dan waktu interview wajib diisi"), // ISO datetime string
  interviewerId: z.string().min(1, "Pewawancara wajib dipilih"),
  notes: z.string().optional(),
});
export type ScheduleInterviewInput = z.infer<typeof scheduleInterviewSchema>;

export const updateOfferFieldsSchema = z.object({
  candidateId: z.string().min(1),
  offeredSalary: z.coerce.number().min(0, "Gaji tidak boleh negatif").optional(),
  startDate: z.string().optional(), // ISO date string
  notes: z.string().optional(),
});
export type UpdateOfferFieldsInput = z.infer<typeof updateOfferFieldsSchema>;
```
